(defun get-keywords-from-file (path)
  (with-temp-buffer
    (insert-file-contents path)
    (delay-mode-hooks (org-mode))
    (org-element-map (org-element-parse-buffer 'element) 'keyword
      (lambda (keyword) (cons (org-element-property :key keyword)
                              (org-element-property :value keyword))))))

(with-temp-buffer
  (dolist (path (delete "./index.org" (directory-files-recursively "." ".org")))
    (insert (format "- [[file:%s][%s]] /%s/\n"
          (file-relative-name path)
          (cdr (assoc "TITLE" (get-keywords-from-file path)))
          (cdr (assoc "CREATED" (get-keywords-from-file path)))
          )))
  (buffer-string))
